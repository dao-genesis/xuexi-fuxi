```bash

# 加：在线 LLM 协填（须 .env 配 API key）
python 课程_闭环.py 环境法学 --feed online --grid

# 加：在线 LLM 出期末模拟卷（写至 _素材/_期末模拟卷.md）

python 课程_闭环.py 环境法学 --rescan

# 单章道喂 + 回填（避免一令误调多章）
python 课程_闭环.py 环境法学 --feed offline --grid --章 3
```
